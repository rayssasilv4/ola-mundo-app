Comandos Git mais utilizados 

 
    git init                      - criar novas pastas no diretorio local
    git clone                     - pegar link no github 
    git add .                     - salvar todas as mudanças feitas 
    git status                    - saber como tá o rolê
    git add                       - para adicionar um arquivo especifico (git add nome_arquivo)
    git commit -m "Mensagem"      - para carregar todas as mudanças numa nova versão
    git log                       - mostra as versões que ja fiz
    git log -n 3                  - mostra alguma versão espesifica (nessa caso a 3ª)
    git log --oneline             - mostra as versões em uma linha apenas 
    git push origin main
    git push origin +main (git push origin main --force)
    git fetch origin main
    git pull origin main
    git pull --rebase origin main
    git cherry-pick
    git branch -a
    git checkout -b
    git checkout
    git merge
    git diff                    - para saber o que mudou
    git restore --staged        - para tirar da area de staged (fica lá depois que eu dou o git add)
    git restore                 - para de fato desfazer algo que fz antes (git restore nome_arquivo)
